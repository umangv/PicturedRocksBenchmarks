\usepackage{pgfplotstable}
% make highlighted tables
% most of this is from https://tex.stackexchange.com/a/42447/21027
\pgfplotstableset{
    color cells/.style={
        col sep=comma,
        string type,
        postproc cell content/.code={%
          \pgfkeysalso{@cell content=\rule{0cm}{2.4ex}\pgfmathtruncatemacro\ucellval{min(100,##1)}\xdef\ucellval{\ucellval}\cellcolor{black!\ucellval}\ifnum\ucellval>50\color{white}\fi##1}%
                },
        columns/0/.style={
            column name={},
            postproc cell content/.code={},
        }
    },
    intersectiontable/.style={color cells,header=has colnames}
}
